## 📝 Frontend Usage

### Option 1: Direct AI Service (No Auth)

```javascript
import { streamAiResponse, exportContent, exportPdf } from '@/services/api'

// Stream content generation
const eventSource = streamAiResponse({
  prompt: 'Write a professional email',
  content_type: 'email',
  tone: 'professional',
  length: 'medium',
  language: 'English'
})

eventSource.onmessage = (event) => {
  const data = JSON.parse(event.data)
  
  if (data.provider) {
    console.log(`Using ${data.provider}`)
  } else if (data.delta) {
    // Append to UI
    appendText(data.delta)
  } else if (data.done) {
    console.log(`Done! ${data.word_count} words`)
    eventSource.close()
  }
}

// Export to HTML
const html = await exportContent(content, 'html', 'email')

// Export to PDF
const pdfBlob = await exportPdf(content, 'resume', 'John Doe')
downloadPdf(pdfBlob, 'resume.pdf')
```

### Option 2: Through Backend (With Auth)

```javascript
import { sendChatMessage, sendChatMessageStream } from '@/services/api'

// Non-streaming
const response = await sendChatMessage({
  prompt: 'Write a blog post',
  contentType: 'blog_post',
  tone: 'professional',
  length: 'long',
  language: 'English',
  sessionId: currentSessionId
})

console.log(response.content)
console.log(response.provider) // Groq, Gemini, etc.
console.log(response.wordCount)

// Streaming
await sendChatMessageStream(
  {
    prompt: 'Write a resume',
    contentType: 'resume',
    tone: 'professional',
    sessionId: currentSessionId
  },
  (data) => {
    // onMessage
    if (data.delta) appendText(data.delta)
  },
  (error) => {
    // onError
    showError(error)
  },
  () => {
    // onComplete
    enableInput()
  }
)
```

## 📊 Complete Example: Resume Generator

```javascript
import { 
  sendChatMessage, 
  exportContent, 
  exportPdf,
  CONTENT_TYPES,
  TONES,
  LENGTHS,
  EXPORT_FORMATS
} from '@/services/api'

async function generateAndExportResume() {
  try {
    // 1. Generate resume
    const response = await sendChatMessage({
      prompt: 'Create a resume for a Senior Software Engineer with 5 years experience',
      contentType: CONTENT_TYPES.RESUME,
      tone: TONES.PROFESSIONAL,
      length: LENGTHS.MEDIUM,
      language: 'English',
      sessionId: currentSessionId
    })
    
    console.log(`Generated by ${response.provider}`)
    console.log(`${response.wordCount} words`)
    
    // 2. Display in UI
    displayResume(response.content)
    
    // 3. Export to HTML
    const htmlExport = await exportContent(
      response.content,
      EXPORT_FORMATS.HTML,
      CONTENT_TYPES.RESUME
    )
    
    // 4. Copy HTML to clipboard
    copyToClipboard(htmlExport.content)
    
    // 5. Export to PDF
    const pdfBlob = await exportPdf(
      response.content,
      CONTENT_TYPES.RESUME,
      'John_Doe'
    )
    
    // 6. Download PDF
    const url = URL.createObjectURL(pdfBlob)
    const a = document.createElement('a')
    a.href = url
    a.download = 'John_Doe_Resume.pdf'
    a.click()
    URL.revokeObjectURL(url)
    
    showSuccess('Resume generated and exported!')
    
  } catch (error) {
    showError(error.message)
  }
}
```

## 🎯 Common Use Cases

### 1. Email Generator with Streaming

```javascript
const eventSource = streamAiResponse({
  prompt: 'Write a meeting invitation for next Monday',
  content_type: 'email',
  tone: 'professional',
  length: 'short'
})

let fullContent = ''

eventSource.onmessage = (event) => {
  const data = JSON.parse(event.data)
  
  if (data.delta) {
    fullContent += data.delta
    updateEditor(fullContent)
  } else if (data.done) {
    eventSource.close()
    enableCopyButton()
  }
}
```

### 2. Blog Post with Custom Instructions

```javascript
const response = await sendChatMessage({
  prompt: 'Write about AI trends in 2026',
  contentType: 'blog_post',
  tone: 'professional',
  length: 'long',
  customInstructions: 'Include statistics and real-world examples',
  sessionId: sessionId
})
```

### 3. Multi-language Social Media

```javascript
const languages = ['English', 'Spanish', 'French']

for (const language of languages) {
  const response = await sendChatMessage({
    prompt: 'Write a product launch announcement',
    contentType: 'social_media',
    tone: 'friendly',
    length: 'short',
    language: language
  })
  
  console.log(`${language}: ${response.content}`)
}
```

### 4. Document Processing

```javascript
// Read uploaded file

  length: 'short'
})
```
